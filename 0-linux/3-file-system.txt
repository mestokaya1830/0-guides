/bin contain executable files
/etc contain system configration files
/opt contain third party app files

help----------------------------------------
ls --help

file and directory list-------------------------
sudo ld -l or ls -al

create directory--------------------------------
mkdir name

remove directory--------------------------------
sudo rm -r name

move directory----------------------------------
sudo mv name path

copy directory----------------------------------
sudo cp name path

FILES------------------------------------------

touch name.txt

remove file--------------------------------
sudo rm -r name

move file----------------------------------
sudo mv name path

copy file----------------------------------
sudo cp name path

rename-------------------------------------
mv name newname

show inside file----------------------------
sudo cat filename

size of directory--------------------------------
du -sh name //size of directory
